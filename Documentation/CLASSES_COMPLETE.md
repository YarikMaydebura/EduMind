## 8. Class Database Schema

```typescript
interface Class {
  id: string;
  name: string;
  theme: Theme;
  rarity: ClassRarity;
  description: string;
  
  // Passive bonuses
  passives: {
    attribute: Attribute;
    percentage: number;
  }[];
  
  // Special abilities (Rare+)
  abilities: {
    id: string;
    name: string;
    description: string;
    effect: AbilityEffect;
  }[];
  
  // Unlock requirements
  requirements: {
    minLevel: number;
    subjectRequirements: {
      subject: Subject;
      minAverage: number;
    }[];
    requiredAchievements: string[];
    specialConditions: SpecialCondition[];
  };
  
  // Evolution
  evolvesFrom: string | null;
  evolvesTo: string[];
  
  // Exclusive skills
  exclusiveSkills: string[];
}

enum Theme {
  FANTASY = 'FANTASY',
  CYBERPUNK = 'CYBERPUNK',
  MILITARY = 'MILITARY',
  SCI_FI = 'SCI_FI',
  ANIME = 'ANIME',
  STEAMPUNK = 'STEAMPUNK',
}

enum ClassRarity {
  COMMON = 'COMMON',
  UNCOMMON = 'UNCOMMON',
  RARE = 'RARE',
  EPIC = 'EPIC',
  LEGENDARY = 'LEGENDARY',
  MYTHIC = 'MYTHIC',
}
```
