## 9. Skill Unlock Condition Examples

```typescript
// ═══════════════════════════════════════════════════════════════════════════════
// SKILL UNLOCK SYSTEM
// ═══════════════════════════════════════════════════════════════════════════════

interface SkillUnlockCondition {
  skillId: string;
  conditions: Condition[];
  operator: 'AND' | 'OR';
}

type Condition = 
  | { type: 'subject_stat'; subject: Subject; minAverage: number }
  | { type: 'achievement'; achievementId: string }
  | { type: 'level'; minLevel: number }
  | { type: 'class'; classIds: string[] }
  | { type: 'class_rarity'; minRarity: ClassRarity }
  | { type: 'battles_won'; count: number }
  | { type: 'mobs_defeated'; count: number }
  | { type: 'homework_completed'; count: number }
  | { type: 'quiz_perfect'; count: number };

// ═══════════════════════════════════════════════════════════════════════════════
// EXAMPLES
// ═══════════════════════════════════════════════════════════════════════════════

const exampleSkills: SkillUnlockCondition[] = [
  // Common skill - easy to get
  {
    skillId: 'phys_001', // Punch
    conditions: [
      { type: 'level', minLevel: 5 },
    ],
    operator: 'AND',
  },
  
  // Uncommon skill - needs subject performance
  {
    skillId: 'magic_101', // Fireball
    conditions: [
      { type: 'subject_stat', subject: 'MATHEMATICS', minAverage: 70 },
      { type: 'subject_stat', subject: 'SCIENCE', minAverage: 65 },
      { type: 'level', minLevel: 10 },
    ],
    operator: 'AND',
  },
  
  // Rare skill - needs achievements too
  {
    skillId: 'magic_201', // Meteor
    conditions: [
      { type: 'subject_stat', subject: 'MATHEMATICS', minAverage: 80 },
      { type: 'subject_stat', subject: 'SCIENCE', minAverage: 75 },
      { type: 'achievement', achievementId: 'quiz_perfect_10' },
      { type: 'level', minLevel: 20 },
    ],
    operator: 'AND',
  },
  
  // Epic skill - needs class requirement
  {
    skillId: 'magic_301', // Apocalypse
    conditions: [
      { type: 'subject_stat', subject: 'MATHEMATICS', minAverage: 85 },
      { type: 'subject_stat', subject: 'SCIENCE', minAverage: 85 },
      { type: 'class_rarity', minRarity: 'EPIC' },
      { type: 'battles_won', count: 100 },
      { type: 'level', minLevel: 35 },
    ],
    operator: 'AND',
  },
  
  // Legendary skill - very strict
  {
    skillId: 'legend_003', // Big Bang
    conditions: [
      { type: 'subject_stat', subject: 'MATHEMATICS', minAverage: 90 },
      { type: 'subject_stat', subject: 'SCIENCE', minAverage: 90 },
      { type: 'class_rarity', minRarity: 'LEGENDARY' },
      { type: 'achievement', achievementId: 'boss_perfect_10' },
      { type: 'level', minLevel: 50 },
    ],
    operator: 'AND',
  },
  
  // Mythic skill - class exclusive
  {
    skillId: 'mythic_005', // Arise (Shadow Monarch)
    conditions: [
      { type: 'class', classIds: ['shadow_monarch'] },
    ],
    operator: 'AND',
  },
```
